119  static bool IsRetryable(int http_code) {
120  if (http_code < 0) return false;
121 
122  switch (http_code) {
123  case HttpStatusCode::INTERNAL_SERVER_ERROR:
124  case HttpStatusCode::SERVICE_UNAVAILABLE:
125  case HttpStatusCode::TOO_MANY_REQUESTS:
126  case HttpStatusCode::BANDWIDTH_LIMIT_EXCEEDED:
127  case HttpStatusCode::REQUEST_TIMEOUT:
128  case HttpStatusCode::AUTHENTICATION_TIMEOUT:
129  case HttpStatusCode::LOGIN_TIMEOUT:
130  case HttpStatusCode::GATEWAY_TIMEOUT:
131  case HttpStatusCode::NETWORK_READ_TIMEOUT:
132  case HttpStatusCode::NETWORK_CONNECT_TIMEOUT:
133  return true;
134  default:
135  return false;
136  }
