148 inline void SubTiles::Skip(std::uint32_t& index) const {
149  if (mask_ == static_cast<std::uint16_t>(-1))
150  return;
151 
152  while (index < count_ && (mask_ & (1 << (index >> shift_))) == 0)
153  ++index;
