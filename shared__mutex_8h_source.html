99  void lock() {

101  if (__ret == EDEADLK)
102  THROW_OR_ABORT(std::system_error(

104  // Errors not handled: EINVAL

113  assert(__ret == 0);

123  // Shared ownership
124 
125  void lock_shared() {
126  int __ret;
127  // We retry if we exceeded the maximum number of read locks supported by
128  // the POSIX implementation; this can result in busy-waiting, but this
129  // is okay based on the current specification of forward progress
130  // guarantees by the standard.
131  do {
132  __ret = pthread_rwlock_rdlock(&_M_rwlock);
133  } while (__ret == EAGAIN);
134  if (__ret == EDEADLK)
135  THROW_OR_ABORT(std::system_error(
136  std::make_error_code(errc::resource_deadlock_would_occur)));
137  // Errors not handled: EINVAL
138  assert(__ret == 0);
139  }
140 
141  bool try_lock_shared() {
142  int __ret = pthread_rwlock_tryrdlock(&_M_rwlock);
143  // If the maximum number of read locks has been exceeded, we just fail
144  // to acquire the lock. Unlike for lock(), we are not allowed to throw
145  // an exception.
146  if (__ret == EBUSY || __ret == EAGAIN)
147  return false;
148  // Errors not handled: EINVAL
149  assert(__ret == 0);
150  return true;
151  }
152 
153  void unlock_shared() { unlock(); }
154 
155  void* native_handle() { return &_M_rwlock; }

161 class shared_mutex_cv {
162  // Based on Howard Hinnant's reference implementation from N2406.
163 
164  // The high bit of _M_state is the write-entered flag which is set to
165  // indicate a writer has taken the lock or is queuing to take the lock.
166  // The remaining bits are the count of reader locks.
167  //
168  // To take a reader lock, block on gate1 while the write-entered flag is
169  // set or the maximum number of reader locks is held, then increment the
170  // reader lock count.
171  // To release, decrement the count, then if the write-entered flag is set
172  // and the count is zero then signal gate2 to wake a queued writer,
173  // otherwise if the maximum number of reader locks was held signal gate1
174  // to wake a reader.
175  //
176  // To take a writer lock, block on gate1 while the write-entered flag is
177  // set, then set the write-entered flag to start queueing, then block on
178  // gate2 while the number of reader locks is non-zero.
179  // To release, unset the write-entered flag and signal gate1 to wake all
180  // blocked readers and writers.
181  //
182  // This means that when no reader locks are held readers and writers get
183  // equal priority. When one or more reader locks is held a writer gets
184  // priority and no more reader locks can be taken while the writer is
185  // queued.
186 
187  // Only locked when accessing _M_state or waiting on condition variables.
188  mutex _M_mut;
189  // Used to block while write-entered is set or reader count at maximum.
190  condition_variable _M_gate1;
191  // Used to block queued writers while reader count is non-zero.
192  condition_variable _M_gate2;
193  // The write-entered flag and reader count.
194  unsigned _M_state;
195 
196  static constexpr unsigned _S_write_entered =
197  1U << (sizeof(unsigned) * __CHAR_BIT__ - 1);
198  static constexpr unsigned _S_max_readers = ~_S_write_entered;
199 
200  // Test whether the write-entered flag is set. _M_mut must be locked.
201  bool _M_write_entered() const { return _M_state & _S_write_entered; }
202 
203  // The number of reader locks currently held. _M_mut must be locked.
204  unsigned _M_readers() const { return _M_state & _S_max_readers; }
205 
206  public:
207  shared_mutex_cv() : _M_state(0) {}
208 
209  ~shared_mutex_cv() { assert(_M_state == 0); }
210 
211  shared_mutex_cv(const shared_mutex_cv&) = delete;
212  shared_mutex_cv& operator=(const shared_mutex_cv&) = delete;
213 
214  // Exclusive ownership
215 
217  void lock() {
218  unique_lock<mutex> __lk(_M_mut);
219  // Wait until we can set the write-entered flag.
220  _M_gate1.wait(__lk, [=] { return !_M_write_entered(); });
221  _M_state |= _S_write_entered;
222  // Then wait until there are no more readers.
223  _M_gate2.wait(__lk, [=] { return _M_readers() == 0; });
224  }
225 
227  bool try_lock() {
228  unique_lock<mutex> __lk(_M_mut, try_to_lock);
229  if (__lk.owns_lock() && _M_state == 0) {
230  _M_state = _S_write_entered;
231  return true;
232  }
233  return false;
234  }
235 
237  void unlock() {
238  lock_guard<mutex> __lk(_M_mut);
239  assert(_M_write_entered());
240  _M_state = 0;
241  // call notify_all() while mutex is held so that another thread can't
242  // lock and unlock the mutex then destroy *this before we make the call.
243  _M_gate1.notify_all();
244  }
245 
246  // Shared ownership
247 
249  void lock_shared() {
250  unique_lock<mutex> __lk(_M_mut);
251  _M_gate1.wait(__lk, [=] { return _M_state < _S_max_readers; });
252  ++_M_state;
253  }
254 
256  bool try_lock_shared() {
257  unique_lock<mutex> __lk(_M_mut, try_to_lock);
258  if (!__lk.owns_lock())
259  return false;
260  if (_M_state < _S_max_readers) {
261  ++_M_state;
262  return true;
263  }
264  return false;
265  }
266 
268  void unlock_shared() {
269  lock_guard<mutex> __lk(_M_mut);
270  assert(_M_readers() > 0);
271  auto __prev = _M_state--;
272  if (_M_write_entered()) {
273  // Wake the queued writer if there are no more readers.
274  if (_M_readers() == 0)
275  _M_gate2.notify_one();
276  // No need to notify gate1 because we give priority to the queued
277  // writer, and that writer will eventually notify gate1 after it
278  // clears the write-entered flag.
279  } else {
280  // Wake any thread that was blocked on reader overflow.
281  if (__prev == _S_max_readers)
282  _M_gate1.notify_one();
283  }
284  }
285 };
