333 #ifndef NDEBUG

335  if (std::numeric_limits<OverflowType>::is_integer &&

425 template <typename T, unsigned int N>
426 auto AlignedBox<T, N>::Distance2(const VectorType& point) const -> ValueType {
427  T distance2 = 0;
428  for (unsigned dim = 0; dim < dimensions; dim++) {
429  if (point[dim] < minimum_[dim]) {
430  T dimDistance = minimum_[dim] - point[dim];
431  distance2 += dimDistance * dimDistance;
432  } else if (point[dim] > maximum_[dim]) {
433  T dimDistance = point[dim] - maximum_[dim];
434  distance2 += dimDistance * dimDistance;
435  }
436  }
437 
438  return distance2;
439 }
