62 bool Wait(std::chrono::milliseconds timeout = std::chrono::seconds(60)) {
63 std::unique_lock<std::mutex> lock(mutex_);
64 bool triggered =
65 condition_.wait_for(lock, timeout, [&] { return signaled_; });
66
67 signaled_ = false;
68 return triggered;
69 }
